60      public void removeSystemProperty(String name) {
61          name = "-D"+name;
62          String nameeq = name+'=';
63          for (Iterator<String> itr = this.iterator(); itr.hasNext();) {
64              String s =  itr.next();
65              if(s.equals(name) || s.startsWith(nameeq))
66                  itr.remove();
67          }
68      }
69  
70      public void setSystemProperty(String name, String value) {
71          removeSystemProperty(name);
72          // index 0 is the executable name
73          add(1,"-D"+name+"="+value);
74      }

101     public static JavaVMArguments of(int pid) throws IOException {
102         String os = System.getProperty("os.name");
103         if("Linux".equals(os))
104             return ofLinux(pid);
105         if("SunOS".equals(os))
106             return ofSolaris(pid);
107         if("Mac OS X".equals(os))
108             return ofMac(pid);
109 
110         throw new UnsupportedOperationException("Unsupported Operating System "+os);
111     }

113     private static JavaVMArguments ofLinux(int pid) throws IOException {
114         pid = resolvePID(pid);
115 
116         String cmdline = readFile(new File("/proc/" + pid + "/cmdline"));
117         JavaVMArguments args = new JavaVMArguments(Arrays.asList(cmdline.split("\0")));
118 
119         // we don't want them inherited
120         args.removeSystemProperty(Daemon.class.getName());
121         args.removeSystemProperty(NetworkServer.class.getName()+".mode");
122         return args;
123     }

130     private static JavaVMArguments ofSolaris(int pid) throws IOException {
131         // /proc shows different contents based on the caller's memory model, so we need to know if we are 32 or 64.
132         // 32 JVMs are the norm, so err on the 32bit side.
133         boolean areWe64 = "64".equals(System.getProperty("sun.arch.data.model"));
134 
135         pid = resolvePID(pid);
136         RandomAccessFile psinfo = new RandomAccessFile(new File("/proc/"+pid+"/psinfo"),"r");
137         try {
138             // see http://cvs.opensolaris.org/source/xref/onnv/onnv-gate/usr/src/uts/common/sys/procfs.h
139             //typedef struct psinfo {
140             //	int	pr_flag;	/* process flags */
141             //	int	pr_nlwp;	/* number of lwps in the process */
142             //	pid_t	pr_pid;	/* process id */
143             //	pid_t	pr_ppid;	/* process id of parent */
144             //	pid_t	pr_pgid;	/* process id of process group leader */
145             //	pid_t	pr_sid;	/* session id */
146             //	uid_t	pr_uid;	/* real user id */
147             //	uid_t	pr_euid;	/* effective user id */
148             //	gid_t	pr_gid;	/* real group id */
149             //	gid_t	pr_egid;	/* effective group id */
150             //	uintptr_t	pr_addr;	/* address of process */
151             //	size_t	pr_size;	/* size of process image in Kbytes */
152             //	size_t	pr_rssize;	/* resident set size in Kbytes */
153             //	dev_t	pr_ttydev;	/* controlling tty device (or PRNODEV) */
154             //	ushort_t	pr_pctcpu;	/* % of recent cpu time used by all lwps */
155             //	ushort_t	pr_pctmem;	/* % of system memory used by process */
156             //	timestruc_t	pr_start;	/* process start time, from the epoch */
157             //	timestruc_t	pr_time;	/* cpu time for this process */
158             //	timestruc_t	pr_ctime;	/* cpu time for reaped children */
159             //	char	pr_fname[PRFNSZ];	/* name of exec'ed file */
160             //	char	pr_psargs[PRARGSZ];	/* initial characters of arg list */
161             //	int	pr_wstat;	/* if zombie, the wait() status */
162             //	int	pr_argc;	/* initial argument count */
163             //	uintptr_t	pr_argv;	/* address of initial argument vector */
164             //	uintptr_t	pr_envp;	/* address of initial environment vector */
165             //	char	pr_dmodel;	/* data model of the process */
166             //	lwpsinfo_t	pr_lwp;	/* information for representative lwp */
167             //} psinfo_t;
168 
169             // see http://cvs.opensolaris.org/source/xref/onnv/onnv-gate/usr/src/uts/common/sys/types.h
170             // for the size of the various datatype.
171 
172             // see http://cvs.opensolaris.org/source/xref/onnv/onnv-gate/usr/src/cmd/ptools/pargs/pargs.c
173             // for how to read this information
174 
175             psinfo.seek(8);
176             if(adjust(psinfo.readInt())!=pid)
177                 throw new IOException("psinfo PID mismatch");   // sanity check
178 
179             /* The following program computes the offset:
180                     #include <stdio.h>
181                     #include <sys/procfs.h>
182                     int main() {
183                       printf("psinfo_t = %d\n", sizeof(psinfo_t));
184                       psinfo_t *x;
185                       x = 0;
186                       printf("%x\n", &(x->pr_argc));
187                     }
188              */
189 
190             psinfo.seek(areWe64?0xEC:0xBC);  // now jump to pr_argc
191             int argc = adjust(psinfo.readInt());
192             long argp = areWe64?adjust(psinfo.readLong()):to64(adjust(psinfo.readInt()));
193             if(LOGGER.isLoggable(FINEST))
194                 LOGGER.finest(String.format("argc=%d,argp=%X",argc,argp));
195 
196             File asFile = new File("/proc/" + pid + "/as");
197             if (areWe64) {
198                 // 32bit and 64bit basically does the same thing, but because the stream position
199                 // is computed with signed long, doing 64bit seek to a position bigger than Long.MAX_VALUE
200                 // requres some real hacking. Hence two different code path.
201                 // (RandomAccessFile uses Java long for offset, so it just can't get to anywhere beyond Long.MAX_VALUE)
202                 FILE fp = LIBC.fopen(asFile.getPath(),"r");
203                 try {
204                     JavaVMArguments args = new JavaVMArguments();
205                     Memory m = new Memory(8);
206                     for( int n=0; n<argc; n++ ) {
207                         // read a pointer to one entry
208                         seek64(fp,argp+n*8);
209                         if(LOGGER.isLoggable(FINEST))
210                             LOGGER.finest(String.format("Seeked to %X",LIBC.ftell(fp)));
211 
212                         m.setLong(0,0); // just to make sure failed read won't result in bogus value
213                         LIBC.fread(m,1,8,fp);
214                         long p = m.getLong(0);
215 
216                         args.add(readLine(fp, p, "argv["+ n +"]"));
217                     }
218                     return args;
219                 } finally {
220                     LIBC.fclose(fp);
221                 }
222             } else {
223                 RandomAccessFile as = new RandomAccessFile(asFile,"r");
224                 try {
225                     JavaVMArguments args = new JavaVMArguments();
226                     for( int n=0; n<argc; n++ ) {
227                         // read a pointer to one entry
228                         as.seek(argp+n*4);
229                         int p = adjust(as.readInt());
230 
231                         args.add(readLine(as, p, "argv["+ n +"]"));
232                     }
233                     return args;
234                 } finally {
235                     as.close();
236                 }
237             }
238         } finally {
239             psinfo.close();
240         }
241     }

243     /**
244      * Seek to the specified position. This method handles offset bigger than {@link Long#MAX_VALUE} correctly.
245      *
246      * @param upos
247      *      This value is interpreted as unsigned 64bit integer (even though it's typed 'long')
248      */
249     private static void seek64(FILE fp, long upos) {
250         LIBC.fseek(fp,0,0); // start at the beginning
251         while(upos<0) {
252             long chunk = Long.MAX_VALUE;
253             upos -= chunk;
254             LIBC.fseek(fp,chunk,1);
255         }
256         LIBC.fseek(fp,upos,1);
257     }

259     /**
260      * {@link DataInputStream} reads a value in big-endian, so
261      * convert it to the correct value on little-endian systems.
262      */
263     private static int adjust(int i) {
264         if(IS_LITTLE_ENDIAN)
265             return (i<<24) |((i<<8) & 0x00FF0000) | ((i>>8) & 0x0000FF00) | (i>>>24);
266         else
267             return i;
268     }
269 
270     private static long adjust(long i) {
271         if(IS_LITTLE_ENDIAN)
272             return (i<<56)
273                     | ((i<<40) & 0x00FF000000000000L)
274                     | ((i<<24) & 0x0000FF0000000000L)
275                     | ((i<< 8) & 0x000000FF00000000L)
276                     | ((i>> 8) & 0x00000000FF000000L)
277                     | ((i>>24) & 0x0000000000FF0000L)
278                     | ((i>>40) & 0x000000000000FF00L)
279                     | (i>>56);
280         else
281             return i;
282     }

285      * int to long conversion with zero-padding.

291     private static String readLine(RandomAccessFile as, int p, String prefix) throws IOException {
292         if(LOGGER.isLoggable(FINEST))
293             LOGGER.finest(String.format("Reading %s at %X",prefix,p));
294 
295         as.seek(to64(p));
296         ByteArrayOutputStream buf = new ByteArrayOutputStream();
297         int ch,i=0;
298         while((ch=as.read())>0) {
299             if((++i)%100==0 && LOGGER.isLoggable(FINEST))
300                 LOGGER.finest(prefix +" is so far "+buf.toString());
301 
302             buf.write(ch);
303         }
304         String line = buf.toString();
305         if(LOGGER.isLoggable(FINEST))
306             LOGGER.finest(prefix+" was "+line);
307         return line;
308     }

310     private static String readLine(FILE as, long p, String prefix) throws IOException {
311         if(LOGGER.isLoggable(FINEST))
312             LOGGER.finest(String.format("Reading %s at %X",prefix,p));
313 
314         seek64(as,p);
315         Memory m = new Memory(1);
316         ByteArrayOutputStream buf = new ByteArrayOutputStream();
317         int i=0;
318         while(true) {
319             if(LIBC.fread(m,1,1,as)==0)   break;
320             byte b = m.getByte(0);
321             if(b==0)    break;
322 
323             if((++i)%100==0 && LOGGER.isLoggable(FINEST))
324                 LOGGER.finest(prefix +" is so far "+buf.toString());
325 
326             buf.write(b);
327         }
328         String line = buf.toString();
329         if(LOGGER.isLoggable(FINEST))
330             LOGGER.finest(prefix+" was "+line);
331         return line;
332     }

335      * Reads the entire file.
336      */
337     private static String readFile(File f) throws IOException {
338         ByteArrayOutputStream baos = new ByteArrayOutputStream();
339         FileInputStream fin = new FileInputStream(f);
340         try {
341             int sz;
342             byte[] buf = new byte[1024];
343 
344             while((sz=fin.read(buf))>=0) {
345                 baos.write(buf,0,sz);
346             }
347 
348             return baos.toString();
349         } finally {
350             fin.close();
351         }

354     /**
355      * Mac support
356      *
357      * See http://developer.apple.com/qa/qa2001/qa1123.html
358      * http://www.osxfaq.com/man/3/kvm_getprocs.ws
359      * http://matburt.net/?p=16 (libkvm is removed from OSX)
360      * where is kinfo_proc? http://lists.apple.com/archives/xcode-users/2008/Mar/msg00781.html
361      *
362      * This code uses sysctl to get the arg/env list:
363      * http://www.psychofx.com/psi/trac/browser/psi/trunk/src/arch/macosx/macosx_process.c
364      * which came from
365      * http://www.opensource.apple.com/darwinsource/10.4.2/top-15/libtop.c
366      *
367      * sysctl is defined in libc.
368      *
369      * PS source code for Mac:
370      * http://www.opensource.apple.com/darwinsource/10.4.1/adv_cmds-79.1/ps.tproj/
371      */
372     private static JavaVMArguments ofMac(int pid) {
373         // local constants
374         final int CTL_KERN = 1;
375         final int KERN_ARGMAX = 8;
376         final int KERN_PROCARGS2 = 49;
377         final int sizeOfInt = Native.getNativeSize(int.class);
378         IntByReference _ = new IntByReference();
379 
380 
381         IntByReference argmaxRef = new IntByReference(0);
382         IntByReference size = new IntByReference(sizeOfInt);
383 
384         // for some reason, I was never able to get sysctlbyname work.
385 //        if(LIBC.sysctlbyname("kern.argmax", argmaxRef.getPointer(), size, NULL, _)!=0)
386         if(LIBC.sysctl(new int[]{CTL_KERN,KERN_ARGMAX},2, argmaxRef.getPointer(), size, NULL, _)!=0)
387             throw new UnsupportedOperationException("Failed to get kernl.argmax: "+LIBC.strerror(Native.getLastError()));
388 
389         int argmax = argmaxRef.getValue();
390         LOGGER.fine("argmax="+argmax);
391 
392         class StringArrayMemory extends Memory {
393             private long offset=0;
394 
395             StringArrayMemory(long l) {
396                 super(l);
397             }
398 
399             int readInt() {
400                 int r = getInt(offset);
401                 offset+=sizeOfInt;
402                 return r;
403             }
404 
405             byte peek() {
406                 return getByte(offset);
407             }
408 
409             String readString() {
410                 ByteArrayOutputStream baos = new ByteArrayOutputStream();
411                 byte ch;
412                 while((ch = getByte(offset++))!='\0')
413                     baos.write(ch);
414                 return baos.toString();
415             }
416 
417             void skip0() {
418                 // skip trailing '\0's
419                 while(getByte(offset)=='\0')
420                     offset++;
421             }
422         }
423         StringArrayMemory m = new StringArrayMemory(argmax);
424         size.setValue(argmax);
425         if(LIBC.sysctl(new int[]{CTL_KERN,KERN_PROCARGS2,resolvePID(pid)},3, m, size, NULL, _)!=0)
426             throw new UnsupportedOperationException("Failed to obtain ken.procargs2: "+LIBC.strerror(Native.getLastError()));
427 
428         
429         /*
430          * Make a sysctl() call to get the raw argument space of the
431          * process.  The layout is documented in start.s, which is part
432          * of the Csu project.  In summary, it looks like:
433          *
434          * /---------------\ 0x00000000
435          * :               :
436          * :               :
437          * |---------------|
438          * | argc          |
439          * |---------------|
440          * | arg[0]        |
441          * |---------------|
442          * :               :
443          * :               :
444          * |---------------|
445          * | arg[argc - 1] |
446          * |---------------|
447          * | 0             |
448          * |---------------|
449          * | env[0]        |
450          * |---------------|
451          * :               :
452          * :               :
453          * |---------------|
454          * | env[n]        |
455          * |---------------|
456          * | 0             |
457          * |---------------| <-- Beginning of data returned by sysctl()
458          * | exec_path     |     is here.
459          * |:::::::::::::::|
460          * |               |
461          * | String area.  |
462          * |               |
463          * |---------------| <-- Top of stack.
464          * :               :
465          * :               :
466          * \---------------/ 0xffffffff
467          */
468 
469         JavaVMArguments args = new JavaVMArguments();
470         int nargs = m.readInt();
471         m.readString(); // exec path
472         for( int i=0; i<nargs; i++) {
473             m.skip0();
474             args.add(m.readString());
475         }
476 
477         // this is how you can read environment variables
478 //        List<String> lst = new ArrayList<String>();
479 //        while(m.peek()!=0)
480 //            lst.add(m.readString());
481 
482         return args;
483     }
